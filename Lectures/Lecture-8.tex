\subsection{Directional Derivative}
\begin{problem}
  If $T(\vec x)$ is a linear transformation, then find the directional derivative of $T(\vec x)$ at $\vec x=\vec a$ in the direction $\vec{v}$.
\end{problem}
\begin{problem}
  If $f:\R^n\to\R$ is defined by $f(\vec x)=\|\vec x\|^2$, find
the directional derivative of $f$ at $\vec x=\vec a$ in the direction $\vec{v}$.
\end{problem}
\textbf{Solution: }
\begin{align*}
D_{\vec v}f(\vec a) &= \lim_{t\to 0} \frac{f(\vec a+t\vec v)-f(\vec a)}{t}\\
&= \lim_{t\to 0} \frac{\|\vec a+t\vec v\|^2-\|\vec a\|^2}{t}\\
&= \lim_{t\to 0} \frac{(\vec a+t\vec v)\cdot(\vec a+t\vec v)-\vec a\cdot \vec a}{t}\\
&= \lim_{t\to 0} \frac{2t\vec a\cdot \vec v+t^2\|\vec v\|^2}{t}\\
&= 2\vec a\cdot \vec v.
\end{align*}
\begin{example}
Let
\[
f(x,y)=
\begin{cases}
\dfrac{x^3}{x^2+y^2}, & (x,y)\neq (0,0),\\[6pt]
0, & (x,y)=(0,0),
\end{cases}
\]
Find the partial derivatives and directional derivatives of $f$ at $(0,0)$.
\end{example}
Partial derivatives at \((0,0)\):
\[
f_x(0,0)=\lim_{h\to 0}\frac{f(h,0)-f(0,0)}{h}
=\lim_{h\to 0}\frac{\frac{h^3}{h^2}-0}{h}
=\lim_{h\to 0}\frac{h}{h}=1.
\]

\[
f_y(0,0)=\lim_{k\to 0}\frac{f(0,k)-f(0,0)}{k}
=\lim_{k\to 0}\frac{0-0}{k}=0.
\]
Directional derivative at \((0,0)\):
Let \(\vec v=(a,b)\neq (0,0)\). Then
\[
D_{\vec v}f(0,0)
=\lim_{t\to 0}\frac{f(ta,tb)-f(0,0)}{t}
=\lim_{t\to 0}\frac{\frac{(ta)^3}{(ta)^2+(tb)^2}}{t}
=\lim_{t\to 0}\frac{\frac{t^3a^3}{t^2(a^2+b^2)}}{t}
=\frac{a^3}{a^2+b^2}.
\]
Equivalently, if \(\vec u=(u_1,u_2)\) is a unit direction, then
\[
D_{\vec u}f(0,0)=\frac{u_1^3}{u_1^2+u_2^2}=u_1^3\neq 0.
\]
But still $f$ is not differentiable at $(0,0)$, see \url{https://math.stackexchange.com/a/3057145/1167703}.\\~\\  
\textbf{Question:} Does the existence of all directional derivatives at a point imply the existence of the total derivative at that point? The answer is no, as shown by the above example. The function $f$ has all directional derivatives at $(0,0)$, but it is not differentiable at $(0,0)$ because the limit in the definition of differentiability does not exist.


\subsection{Tangent Space for $\mathbb R^2$ (Optional)}
For a \emph{fixed} direction $v$, the directional derivative at $p$ defines a map
\[
D_v\big|_p: C^\infty(\mathbb{R}^2)\longrightarrow \mathbb{R},
\qquad
f\longmapsto D_v f(p),
\]
which is linear. Moreover, it satisfies the Leibniz rule
\[
D_v(fg)(p)=f(p)\,D_v g(p)+g(p)\,D_v f(p),
\]
so $D_v|_p$ is a \emph{derivation at $p$} (the ``tangent vector as operator'' viewpoint).

\begin{definition}
A derivation is a linear map
\[
X : C^\infty(\mathbb R^2) \to \mathbb{R}
\]
acting on smooth functions, satisfying the
Leibniz rule:
\[
X(fg) = X(f)g + fX(g).
\]
It generalizes the concept of a tangent vector as an operator, mapping a
function to its directional derivative at a specific point or over a region.
\end{definition}

\medskip
The tangent space is \emph{not} built from one fixed operator $D_v|_p$; rather it is built from the collection of \emph{all} such derivations at $p$. A clean definition is
\[
T_p\mathbb{R}^2
:=\Bigl\{\,X:C^\infty(\mathbb{R}^2)\to\mathbb{R}\ \Bigm|\ 
X \text{ is linear and } X(fg)=f(p)X(g)+g(p)X(f)\Bigr\}.
\]
Elements of $T_p\mathbb{R}^2$ are operators that differentiate functions at $p$.
Each direction $v=(v_1,v_2)\in\mathbb{R}^2$ produces an element
\[
v\longmapsto D_v\big|_p\in T_p\mathbb{R}^2.
\]
\medskip
Define the standard derivations
\[
\left.\frac{\partial}{\partial x}\right|_p(f)=\frac{\partial f}{\partial x}(p),
\qquad
\left.\frac{\partial}{\partial y}\right|_p(f)=\frac{\partial f}{\partial y}(p).
\]
Then for $v=(v_1,v_2)$,
\[
D_v\big|_p
=
v_1\left.\frac{\partial}{\partial x}\right|_p
+
v_2\left.\frac{\partial}{\partial y}\right|_p,
\]
so $\Bigl\{\left.\frac{\partial}{\partial x}\right|_p,\left.\frac{\partial}{\partial y}\right|_p\Bigr\}$
is a basis of $T_p\mathbb{R}^2$.
\medskip
A tangent vector at $p$ can be viewed as an operator that takes a smooth function $f$ 
and returns its directional rate of change at $p$. The tangent space $T_p\mathbb{R}^2$ 
is the vector space of all such derivative-operators (derivations), 
and in coordinates it has basis 
$\left.\frac{\partial}{\partial x}\right|_p,\left.\frac{\partial}{\partial y}\right|_p$. See \url{https://math.stackexchange.com/q/1118419/1167703} to get idea why are derivations useful for defining tangent vectors. 
\begin{example}
  Let $f(x,y)=3x-2y$ and $p=(1,1)$.
For $v=(1,0)$ and $w=(0,1)$,
\[
D_v\big|_p(f)=\frac{\partial f}{\partial x}(p)=3,
\qquad
D_w\big|_p(f)=\frac{\partial f}{\partial y}(p)=-2.
\]
For $u=(2,5)$,
\[
D_u\big|_p(f)=2\cdot 3+5\cdot(-2)=6-10=-4.
\]
Equivalently,
\[
D_u\big|_p
=
2\left.\frac{\partial}{\partial x}\right|_p
+
5\left.\frac{\partial}{\partial y}\right|_p.
\]
\end{example}
\begin{example}
Let $f(x,y)=x^2+y^2$ and $p=(1,2)$. Then
\[
f_x=2x,\qquad f_y=2y
\quad\Rightarrow\quad
f_x(p)=2,\ \ f_y(p)=4.
\]
For $v=(3,-1)$,
\[
D_v\big|_p(f)=3\cdot 2+(-1)\cdot 4=6-4=2,
\qquad\text{hence}\qquad
D_{(3,-1)}\big|_p
=
3\left.\frac{\partial}{\partial x}\right|_p
-
\left.\frac{\partial}{\partial y}\right|_p.
\]
\end{example}
\begin{example}
From the previous example, for $v=(1,0)$ and $w=(0,1)$,
\[
D_v\big|_p(f)=2,\qquad D_w\big|_p(f)=4.
\]
Since $v+w=(1,1)$,
\[
D_{v+w}\big|_p(f)=1\cdot 2+1\cdot 4=6,
\qquad\text{and indeed}\qquad
D_{v+w}\big|_p(f)=D_v\big|_p(f)+D_w\big|_p(f).
\]
\end{example}
\begin{example}
Let $f(x,y)=x$, $g(x,y)=y^2$, $p=(1,2)$, and $v=(3,-1)$.
Then
\[
D_v\big|_p(f)=3,
\qquad
D_v\big|_p(g)=(-1)\cdot (2y)\big|_{(1,2)}=-4.
\]
For the product $fg=xy^2$, compute
\[
(xy^2)_x=y^2,\qquad (xy^2)_y=2xy,
\]
so at $(1,2)$ we have $D(fg)(1,2)=(4,4)$ and hence
\[
D_v\big|_p(fg)=(4,4)\cdot(3,-1)=12-4=8.
\]
On the other hand,
\[
f(p)\,D_v\big|_p(g)+g(p)\,D_v\big|_p(f)
=(1)(-4)+(4)(3)=-4+12=8,
\]
\end{example}
~\\
Tangent spaces let us define:
\begin{enumerate}
  \item Derivatives
  \item Gradients
  \item Vector fields
  \item Flows
  \item Differential equations
\end{enumerate}

\subsection{Total Derivative}
\begin{definition}
A function $f:\mathbb{R}\to\mathbb{R}$ is called \emph{differentiable at $x=a$}
if there exists a real number $L$ such that
\[
\lim_{h\to 0}\frac{f(a+h)-f(a)-Lh}{h}=0.
\]
Then the number $L$ is called the \emph{derivative} of $f$ at $x=a$.
\end{definition}

The definition indicates that,
$$f(a+h)=f(a)+Lh+o(h)$$
So, near $x=a$, the function $f$ behaves like a linear function with slope $L$, $f(a+h) \approx f(a) + Lh=\text{base point}+\text{linear correction}$. The derivative only captures the linear correction part. The constant term just shifts the graph vertically.

\textbf{Remark:} The notation $o(h)$ means that 
the error term $E(h)=f(a+h)-f(a)-Lh$ is small compared 
to $h$ as $h \to 0$. More precisely, it means that 
$\lim_{h\to 0} E(h)/h=0$. In other words, $E(h)$ goes to 
zero faster than $h$ does. Formally, we say a function $E(h)$
is $o(h)$ if $$\lim_{h\to 0} E(h)/h=0.$$ In one variable 
$h^2=o(h)$, $h^3=o(h)$, but $2h$ is not $o(h)$ because $\lim_{h\to 0} 2h/h=2 \neq 0$.

\begin{definition}
A function $f:\mathbb{R}^n \to \mathbb{R}^m$ 
is called \emph{differentiable at $x=\vec a$}
if there exists a linear transformation 
$T_{\vec a}:\mathbb{R}^n \to \mathbb{R}^m$
such that
\[
\lim_{\vec h \to 0}
\frac{
\left\| f(\vec a+\vec h) - f(\vec a) - T_{\vec a}(\vec h) \right\|
}{
\|\vec h\|
}
=0.
\]
Then $T_{\vec a}$ is called the \emph{total derivative} at $x=\vec a$,
and is often denoted by $Df(\vec a)$.
\end{definition}
Again, this says that $Df(\vec a)$ is the best linear approximation to the function $f-f(\vec a)$ at $\vec a$,
in the sense that the difference $f(\vec a+\vec h)-f(\vec a)-Df(\vec a)(\vec h)$ is small compared to $\|\vec h\|$ as $\vec h \to 0$.

\begin{proposition}
Suppose there exist two linear transformations
\[
T_{\vec a}^{(1)},\; T_{\vec a}^{(2)} : \mathbb{R}^n \to \mathbb{R}^m
\]
such that
\[
\lim_{\|\vec h\|\to 0}
\frac{
\|f(\vec a+\vec h)-f(\vec a)-T_{\vec a}^{(j)}(\vec h)\|
}{
\|\vec h\|
}
=0,
\qquad j=1,2.
\]
Prove that
\[
T_{\vec a}^{(1)}=T_{\vec a}^{(2)}.
\]
\end{proposition}
\begin{proof}
Since both limits are zero,
\[
T_{\vec a}^{(2)}(\vec h)-T_{\vec a}^{(1)}(\vec h)
=
\{f(\vec a+\vec h)-f(\vec a)-T_{\vec a}^{(1)}(\vec h)\}
-
\{f(\vec a+\vec h)-f(\vec a)-T_{\vec a}^{(2)}(\vec h)\}.
\]
Taking norms and applying the triangle inequality gives
\[
\|(T_{\vec a}^{(2)}-T_{\vec a}^{(1)})(\vec h)\|
\le
\|f(\vec a+\vec h)-f(\vec a)-T_{\vec a}^{(1)}(\vec h)\|
+
\|f(\vec a+\vec h)-f(\vec a)-T_{\vec a}^{(2)}(\vec h)\|.
\]
Dividing by $\|\vec h\|$ and taking the limit yields
\[
\lim_{\|\vec h\|\to 0}
\frac{
\|(T_{\vec a}^{(2)}-T_{\vec a}^{(1)})(\vec h)\|
}{
\|\vec h\|
}
=0.
\]
Let
\[
L := T_{\vec a}^{(2)}-T_{\vec a}^{(1)}.
\]
Then $L$ is linear and
\[
\lim_{\|\vec h\|\to 0}
\frac{\|L(\vec h)\|}{\|\vec h\|}=0.
\]
This only tells us something about vectors $\vec h$ near $\vec 0$.
It does not immediately tell you what $L(\vec h)$ is for a fixed nonzero $\vec h$. Uniqueness needs a statement like the operator is zero on every vector.
The trick plug in $\alpha \vec h$ where $\alpha \to 0$ is exactly how you turn a local limit statement into a global conclusion using linearity.\\
Let $\vec h \neq 0$ be fixed.
For any scalar $\alpha \to 0$, we have $\alpha \vec h \to 0$.
Thus the limit applies:
\[
0=
\lim_{\alpha\to 0}
\frac{
\|L(\alpha \vec h)\|
}{
\|\alpha \vec h\|
}.
\]
By linearity,
\[
L(\alpha \vec h)=\alpha L(\vec h),
\]
so
\[
\frac{
\|L(\alpha \vec h)\|
}{
\|\alpha \vec h\|
}
=
\frac{
\|\alpha L(\vec h)\|
}{
|\alpha|\,\|\vec h\|
}
=
\frac{
\|L(\vec h)\|
}{
\|\vec h\|
}.
\]
\[
0=
\lim_{\alpha\to 0}
\frac{\|L(\alpha \vec h)\|}{\|\alpha \vec h\|}
=
\lim_{\alpha\to 0}
\frac{\|L(\vec h)\|}{\|\vec h\|}.
\]
Notice the quantity inside the limit is constant in $\alpha$.
The limit of a constant sequence/function is just the constant itself. So
\[
\frac{\|L(\vec h)\|}{\|\vec h\|}=0 \implies \|L(\vec h)\|=0.
\]
And the only vector with zero norm is the zero vector. Hence $L(\vec h)=0$.
Since the above holds for every nonzero vector $\vec h$, $L \equiv 0$.
Thus
\[
T_{\vec a}^{(2)}=T_{\vec a}^{(1)}.
\]
\end{proof}

%  Example 1
\begin{example}
Consider the function $f:\R^2 \to \R$ defined by
$$
f(x,y) = x^2-y^2
$$
Find the derivative of $f$ at the point $\begin{pmatrix}
2 \\ 3
\end{pmatrix}$, find the value of $Df\begin{pmatrix}
2 \\ 3
\end{pmatrix}$.
\end{example}
\textbf{Solution.} Let $$Df\begin{pmatrix}2 \\ 3
\end{pmatrix}=\begin{pmatrix}
  a & b
\end{pmatrix}$$
Using the definition of differentiability, let $h=(h_1,h_2)\to(0,0)$. Then
\begin{align*}
&\lim_{\|\vec h\|\to 0}\frac{\left\|f\left(\begin{pmatrix}
2\\3
\end{pmatrix}+\begin{pmatrix}h_1\\h_2\end{pmatrix}\right)-f\begin{pmatrix}2\\3\end{pmatrix}-\begin{pmatrix}a&b\end{pmatrix}\begin{pmatrix}h_1\\h_2\end{pmatrix}\right\|}{\|\vec h\|}\\
=&
\lim_{\|\vec h\|\to 0}
\frac{\left\|f\begin{pmatrix}2+h_1\\3+h_2\end{pmatrix}-f\begin{pmatrix}2\\3\end{pmatrix}-a h_1-b h_2\right\|}{\sqrt{h_1^2+h_2^2}}.  
\end{align*}
Compute:
\[
f(2+h_1,3+h_2)-f(2,3)
=(2+h_1)^2-(3+h_2)^2-\bigl(2^2-3^2\bigr).
\]
and
\[
(2+h_1)^2-(3+h_2)^2-2^2+3^2
= (4+4h_1+h_1^2)-(9+6h_2+h_2^2)-4+9
=4h_1-6h_2+h_1^2-h_2^2.
\]
So the numerator becomes
\[
\left|(4-a)h_1-(6+b)h_2+h_1^2-h_2^2\right|.
\]
To make the linear terms vanish (\textcolor{red}{why?}), choose
\[
4-a=0\Rightarrow a=4,\qquad -(6+b)=0\Rightarrow b=-6.
\]
Then the limit reduces to
\[
\lim_{(h_1,h_2)\to(0,0)}\frac{|h_1^2-h_2^2|}{\sqrt{h_1^2+h_2^2}}
\le
\lim_{(h_1,h_2)\to(0,0)}\frac{h_1^2+h_2^2}{\sqrt{h_1^2+h_2^2}}
=
\lim_{(h_1,h_2)\to(0,0)}\sqrt{h_1^2+h_2^2}=0.
\]
Hence $f$ is differentiable at $(2,3)$ and
\[
Df\begin{pmatrix}2\\3\end{pmatrix}=\begin{pmatrix}4&-6\end{pmatrix}.
\]

\begin{example}
For $f:\mathbb{R}^2\to\mathbb{R}^2$ given by
\[
f(x,y)=\binom{xy}{y^2},
\]
find the value of $Df\begin{pmatrix}
a \\ b
\end{pmatrix}$.
\end{example}
\textbf{Solution:} For $f:\mathbb{R}^2\to\mathbb{R}^2$ given by
\[
f(x,y)=\binom{xy}{y^2},
\]
the derivative at a point $\vec a=\begin{pmatrix}a\\b\end{pmatrix}$ is the linear map $Df\begin{pmatrix}
a\\b
\end{pmatrix}$ satisfying
\[
f\begin{pmatrix}a+h\\b+k\end{pmatrix}-f\begin{pmatrix}a\\b\end{pmatrix}=Df_{\vec a}\binom{h}{k}+o(\|(h,k)\|).
\]
We will see this definition can be used to compute the derivative without using the limit definition directly.
Compute the increment:
\[
f\begin{pmatrix}a+h\\b+k\end{pmatrix}=\binom{(a+h)(b+k)}{(b+k)^2}
=\binom{ab+ak+bh+hk}{b^2+2bk+k^2}.
\]
So
\[
f\begin{pmatrix}a+h\\b+k\end{pmatrix}-f\begin{pmatrix}a\\b\end{pmatrix}=\binom{ak+bh+hk}{2bk+k^2}.
\]
The linear part in $(h,k)$ is
\[
\binom{bh+ak}{2bk},
\]
and the remainder is
\[
\binom{hk}{k^2},
\]
\textcolor{red}{(you can skip to verify the error term is $o(\|(h,k)\|)$, but we will verify it for the sake of completeness)} whose norm is
\\
~\\
which is $o(\|(h,k)\|)$ because $|hk|\le \tfrac12(h^2+k^2)$ and $|k^2|\le h^2+k^2$.
\[
\boxed{
\begin{aligned}
\text{Claim:}\quad &\sqrt{x^2+y^2}\le |x|+|y|.\\[4pt]
\text{Proof:}\quad 
&(|x|+|y|)^2
= x^2+y^2+2|x||y|
\ge x^2+y^2.\\[4pt]
&\text{Since both sides are nonnegative, taking square roots gives}\\[4pt]
&\sqrt{x^2+y^2}\le |x|+|y|.
\end{aligned}
}
\]
\[
\|R(h,k)\|=\sqrt{(hk)^2+(k^2)^2}\le |hk|+|k^2|
\le \frac{h^2+k^2}{2}+(h^2+k^2)=\frac{3}{2}(h^2+k^2)=\frac{3}{2}\|(h,k)\|^2.
\]
\[
\frac{\|R(h,k)\|}{\|(h,k)\|}\le \frac{3}{2}\|(h,k)\|\to 0,
\]
Therefore,
\[
Df_{\vec a}\binom{h}{k}=\binom{bh+ak}{2bk},
\]
So the matrix of $Df_{\vec a}$ is
\[
Df_{\vec a}=
\begin{pmatrix}
b & a\\
0 & 2b
\end{pmatrix}.
\]

\end{remark}
