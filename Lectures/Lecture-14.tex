\section{Lecture-14}
\subsection{Double Integrals (contd.)}
\begin{example}
Prove the Gaussian integral:
\[
   \int_{-\infty}^{\infty} e^{-x^2}\:dx=\sqrt{\pi}.
\]
\end{example}

\begin{example}
Evaluate using polar coordinates:
$$
   \iint_D \arctan\left(\frac{y}{x}\right),dx,dy,
$$
where D is the region bounded by $x^2+y^2=1, x^2+y^2=4$, and the lines $y=x,y=\sqrt3x$.\\ 
\textbf{Solution:} $\boxed{\frac{7\pi^2}{192}}$
\end{example}


\begin{example}
Evaluate the double integral by changing variables:
$$
   \iint_D \sqrt{xy}\:dx\:dy,
$$
where $D$ is the region in the first quadrant bounded by $y^2=3x, y^2=6x, xy=4, xy=9$.\\
\textbf{Solution:} $\boxed{\frac{38\ln{2}}{9}}$  

\end{example}

\begin{example}
    Let $V$ be bounded by $y=x, y=2x, z=h, z=x^2+y^2$ and $z=2(x^2+y^2)$ where $x,y,z\geq 0$. Find the volume of this solid.\\ 
\textbf{Solution:} $\boxed{\frac{h^2}{8}\left(\arctan 2-\frac{\pi}{4}\right)}$ 
\end{example}

\begin{example}
   Find the volume of the upper hemisphere of the sphere $x^2+y^2+z^2=1$ that lies above the $xy$-plane.\\
\textbf{Solution:} $\boxed{\frac{2\pi}{3}}$
\end{example}

\begin{example}
Evaluate the triple integral using a change of variables:
$$
    \iiint_V \frac{xy}{z}\:dx\:dy\:dz
$$
where
$$
V=\{(x,y,z):x,y,z\ge0,3x\le y\le4x,1\le xy\le2,z\le x^2+y^2\le3z\}.
$$


\end{example}
